# 系统设置(模型管理 + 深浅色主题)实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** ccswitch 式模型管理(自动从 Ark `/models` 拉取可用模型 + 按 AI 功能分配)+ 全局 light/dark 主题切换。模型分配 DB 持久化、网关读 DB(env 兜底);主题 localStorage 记忆、CSS 变量 + ECharts 跟随。

**Tech Stack:** 后端 Node.js + Express + TS + pg + zod + vitest/supertest;前端 Vue3 + Vite + TS + ECharts。

**Spec:** `docs/superpowers/specs/2026-07-21-系统设置-design.md`

```ts
import { query } from '../db/pool.js';
import { config } from '../config.js';
export type AiFeature = 'chat' | 'poster' | 'extraction';
const ENV_FALLBACK: Record<AiFeature, string> = {
  chat: config.doubaoModel, poster: config.posterModel, extraction: config.extractModel,
};
export async function getFeatureModel(feature: AiFeature): Promise<string> {
  const r = (await query<{ model_id: string }>(
    'SELECT model_id FROM ai_feature_config WHERE feature=$1', [feature])).rows[0];
  return r?.model_id || ENV_FALLBACK[feature];
}
```

```ts
function classifyKind(taskType: string[] = []): 'chat' | 'image' | 'other' {
  if (taskType.some(t => t === 'TextToImage' || t === 'ImageToImage')) return 'image';
  if (taskType.some(t => t === 'TextGeneration' || t === 'VisualQuestionAnswering')) return 'chat';
  return 'other';
}
export async function refreshModels() {
  if (!config.doubaoApiKey) throw new Error('AI 未配置');
  const r = await fetch(`${config.doubaoBaseUrl}/models`, { headers: { Authorization: `Bearer ${config.doubaoApiKey}` } });
  if (!r.ok) throw new Error(`Ark /models ${r.status}`);
  const j = await r.json() as { data: any[] };
  const usable = (j.data || []).filter(m => m.status !== 'Shutdown' && m.status !== 'Retiring');
  for (const m of usable) {
    const kind = classifyKind(m.task_type);
    await query(
      `INSERT INTO ai_model (model_id, label, kind, status, task_type, fetched_at, enabled)
       VALUES ($1,$2,$3,$4,$5,now(),true)
       ON CONFLICT (model_id) DO UPDATE SET label=EXCLUDED.label, kind=EXCLUDED.kind, status=EXCLUDED.status, task_type=EXCLUDED.task_type, fetched_at=now()`,
      [m.id, m.name || m.id, kind, m.status || '', JSON.stringify(m.task_type || [])],
    );
  }
  return { fetched: usable.length, total: usable.length };
}
```

```ts
import { ref } from 'vue';
export type ThemeKey = 'light' | 'dark';
const theme = ref<ThemeKey>((localStorage.getItem('theme') as ThemeKey) || 'light');
const apply = (k: ThemeKey) => { document.documentElement.dataset.theme = k; localStorage.setItem('theme', k); };
apply(theme.value);
export const useTheme = () => ({
  theme,
  toggle: () => { theme.value = theme.value === 'dark' ? 'light' : 'dark'; apply(theme.value); },
});
```
