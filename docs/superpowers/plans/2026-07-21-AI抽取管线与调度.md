```ts
export const isValidDate = (v: unknown) => typeof v === 'string' && /^\d{4}[-/]\d{2}[-/]\d{2}$/.test(v.trim());
export const typeCheck = (type: string, val: unknown): boolean => { /* 原 sync 逻辑 */ };
// 清洗数值(去 ¥ , ， 空格)并强转;返回 {ok, value} 或 {ok:false, error}
export const coerceMetric = (type: string, val: unknown): { ok: true; value: any } | { ok: false; error: string } => {
  if (isEmpty(val)) return { ok: true, value: undefined }; // 空值不写
  if (type === 'number' || type === 'int') {
    const cleaned = String(val).replace(/[¥￥,，\s]/g, '');
    const n = Number(cleaned);
    if (isNaN(n)) return { ok: false, error: `类型错误(${type})` };
    return { ok: true, value: type === 'int' ? Math.trunc(n) : n };
  }
  if (type === 'date') return isValidDate(val) ? { ok: true, value: String(val).slice(0,10) } : { ok: false, error: '日期格式错误' };
  return { ok: true, value: String(val) };
};
```

```ts
export class NotConfigured extends Error { constructor() { super('AI 未配置'); } }
export async function callDoubaoJson(messages: ChatMessage[]): Promise<any> {
  if (!config.doubaoApiKey) throw new NotConfigured();
  const url = `${config.doubaoBaseUrl}/chat/completions`;
  const r = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${config.doubaoApiKey}` },
    body: JSON.stringify({ model: config.extractModel, messages, response_format: { type: 'json_object' }, temperature: 0 }),
  });
  if (!r.ok) throw new Error(`豆包 API ${r.status}: ${(await r.text()).slice(0, 300)}`);
  const resp = await r.json();
  const content = resp.choices?.[0]?.message?.content;
  if (!content) throw new Error('AI 未返回内容');
  try { return JSON.parse(content); }
  catch { throw new Error('AI 返回非合法 JSON'); }
}
```

```ts
// data.sheets[sheetKey].cellData = {row:{col:{v}}}
function serializeSheet(sheet: any): { tsv: string; rowsIn: number } {
  const cd = sheet.cellData || {};
  const rowIdx = Object.keys(cd).map(Number).sort((a,b)=>a-b);
  if (!rowIdx.length) return { tsv: '', rowsIn: 0 };
  const lines: string[] = [];
  let rowsIn = 0;
  for (const r of rowIdx) {
    if (r >= 400) break;
    const row = cd[r];
    const cols = Object.keys(row).map(Number);
    const maxC = Math.max(...cols);
    const cells: string[] = [];
    for (let c = 0; c <= maxC; c++) cells.push(row[c]?.v == null ? '' : String(row[c].v));
    if (r === 0 || cells.some(x => x !== '')) { lines.push(cells.join('\t')); if (r > 0) rowsIn++; }
  }
  return { tsv: lines.join('\n'), rowsIn };
}
```

```ts
import cron from 'node-cron';
import { config } from '../config.js';
import { query } from '../db/pool.js';
import { extractWorkbook } from './extraction.service.js';

export async function runScheduledExtraction() {
  const period = periodOf(new Date()); // YYYY-MM
  const { rows } = await query(
    `SELECT id FROM workbook WHERE period=$1 AND deleted_at IS NULL
     AND (last_extracted_at IS NULL OR updated_at > last_extracted_at)`, [period]);
  for (const w of rows.rows ?? rows) {
    try { await withTimeout(extractWorkbook(w.id, { source: 'scheduled', userId: null }), config.extractTimeoutMs); }
    catch (e) { console.error('[extraction] scheduled fail wb=%s:', w.id, (e as Error).message); }
  }
}
export function startScheduler() {
  if (process.env.NODE_ENV === 'test') return;
  cron.schedule(config.extractCron, () => { runScheduledExtraction().catch(e => console.error('[extraction] cron:', e)); });
}
```
